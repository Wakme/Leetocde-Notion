    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int n = nums1.length, m = nums2.length;
        if(n > m) return findMedianSortedArrays(nums2, nums1);
        int l = 0, lma = (n + m + 1) / 2, r = n;
        while(l < r){
            int mid = l + r + 1 >> 1;
            int l1 = mid, l2 = lma - l1;
            if(nums1[l1 - 1] <= nums2[l2]) l = mid;
            else r = mid - 1;
        }
        int l1 = l, l2 = lma - l;
        int nums1lma = (l1 == 0 ? Integer.MIN_VALUE: nums1[l1 - 1]);
        int nums1rmi = (l1 == n ? Integer.MAX_VALUE: nums1[l1]);
        int nums2lma = (l2 ==   0 ? Integer.MIN_VALUE: nums2[l2 - 1]);
        int nums2rmi = (l2 == m ? Integer.MAX_VALUE: nums2[l2]);
        int ma = Math.max(nums1lma, nums2lma);
        int mi = Math.min(nums1rmi, nums2rmi);
        if((n + m) % 2 == 0) return (ma + mi) / 2.0;
        return ma * 1.0;
    }
